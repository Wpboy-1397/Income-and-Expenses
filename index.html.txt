function checkAppPassword() {
  const pwd = document.getElementById('appPassword').value;
  const errDiv = document.getElementById('loginErr');
  if(!pwd) return alert("กรุณากรอกรหัสผ่าน");
  
  google.script.run.withSuccessHandler(isValid => {
    if(isValid) {
      document.getElementById('loginOverlay').style.display = 'none';
      updateYearLists(); 
      updateItemDropdown();
    } else {
      errDiv.style.display = 'block';
      document.getElementById('appPassword').value = '';
    }
  }).verifyPassword(pwd);
}

function go(p) {
  document.querySelectorAll('.page').forEach(el => el.classList.remove('active'));
  document.querySelectorAll('.nav button').forEach(el => el.classList.remove('active'));
  document.getElementById(p + 'Page').classList.add('active');
  document.getElementById('nav-' + p).classList.add('active');
}

function updateYearLists() {
  google.script.run.withSuccessHandler(years => {
    const html = years.map(y => `<option value="${y}">${y}</option>`).join('') + '<option value="all">ทุกปี</option>';
    document.querySelectorAll('.y-list').forEach(s => s.innerHTML = html);
  }).getYearLists();
}

function loadV() {
  const mFrom = document.getElementById('vMFrom').value;
  const mTo = document.getElementById('vMTo').value;
  const y = document.getElementById('vY').value;
  const item = document.getElementById('vItem').value;

  if (mFrom !== 'all' && mTo !== 'all' && parseInt(mFrom) > parseInt(mTo)) {
    alert("กรุณาเลือกช่วงเดือนให้ถูกต้อง (เดือนเริ่มต้นต้องไม่มากกว่าเดือนสิ้นสุด)");
    return;
  }

  google.script.run.withSuccessHandler(res => {
    const area = document.getElementById('vArea'), msg = document.getElementById('noMsgV');
    if (!res.summaryData || res.summaryData.length === 0) { 
      area.style.display = 'none'; msg.style.display = 'block'; 
      var mTxt = (mFrom === 'all' || mTo === 'all') ? 'ทุกเดือน' : `${mNames[mFrom]} - ${mNames[mTo]}`;
      msg.innerHTML = `ไม่พบข้อมูลของ: <b>${mTxt} ปี ${y}</b> (${item})`; return; 
    }
    msg.style.display = 'none'; area.style.display = 'block';
    var headTxt = (mFrom === 'all' || mTo === 'all') ? 'ทุกเดือน' : `${mNames[mFrom]} - ${mNames[mTo]}`;
    document.getElementById('vHead').innerHTML = `ข้อมูลของ: ${headTxt} ปี ${y} (${item})`;
    document.getElementById('vBal').innerText = (res.income - res.expense).toLocaleString();
    document.getElementById('vInc').innerText = res.income.toLocaleString();
    document.getElementById('vExp').innerText = res.expense.toLocaleString();
    document.getElementById('vBody').innerHTML = res.summaryData.map(s => 
      `<tr><td>${s.name}</td><td style="text-align:right; color:${s.type === 'รายรับ' ? 'green' : 'red'}">${s.amount.toLocaleString()}</td></tr>`).join('');
    drawCharts(res.income, res.expense, res.summaryData);
  }).getGroupSummaryRange(mFrom, mTo, y, item);
}

function drawCharts(inc, exp, summary) {
  new google.visualization.PieChart(document.getElementById('pChart')).draw(google.visualization.arrayToDataTable([['Type', 'Amt'], ['รายรับ', inc], ['รายจ่าย', exp]]), { pieHole: 0.4, colors: ['#2e7d32', '#c62828'], chartArea:{width:'90%',height:'80%'}, legend:'bottom' });
  const barData = [['รายการ', 'เงิน', { role: 'style' }, { role: 'annotation' }]];
  const colors = ['#3949ab', '#66bb6a', '#ffa726', '#ef5350', '#26c6da', '#ab47bc'];
  summary.forEach((s, idx) => barData.push([s.name, s.amount, colors[idx % colors.length], s.amount.toLocaleString()]));
  new google.visualization.ColumnChart(document.getElementById('colChart')).draw(google.visualization.arrayToDataTable(barData), { legend:'none', chartArea:{width:'80%', height:'70%'} });
}

function loadR() {
  const item = document.getElementById('rItem').value;
  const mFrom = document.getElementById('rMFrom').value;
  const mTo = document.getElementById('rMTo').value;
  const y = document.getElementById('rY').value;

  if (mFrom !== 'all' && mTo !== 'all' && parseInt(mFrom) > parseInt(mTo)) {
    alert("กรุณาเลือกช่วงเดือนให้ถูกต้อง");
    return;
  }

  google.script.run.withSuccessHandler(data => {
    const area = document.getElementById('reportArea'), msg = document.getElementById('noMsgR');
    var mTxt = (mFrom === 'all' || mTo === 'all') ? 'ทุกเดือน' : `${mNames[mFrom]} - ${mNames[mTo]}`;
    if (!data || data.length === 0) { area.style.display = 'none'; msg.style.display = 'block'; msg.innerHTML = `ไม่พบข้อมูลของ: <b>${mTxt} ปี ${y}</b> (${item})`; return; }
    msg.style.display = 'none'; area.style.display = 'block';
    document.getElementById('rHead').innerText = `ข้อมูล ${item} (${mTxt} ${y})`;
    let inc = 0, exp = 0;
    document.getElementById('rBody').innerHTML = data.map(r => { 
      if(r.type === 'รายรับ') inc += r.amount; else exp += r.amount;
      return `<tr><td>${r.date}</td><td>${r.item}</td><td>${r.note}</td><td style="text-align:right; color:${r.type === 'รายรับ' ? 'green' : 'red'}">${r.amount.toLocaleString()}</td></tr>`; 
    }).join('');
    document.getElementById('rDetailAmt').innerHTML = `<span style="color:green">รับ: ${inc.toLocaleString()}</span> | <span style="color:red">จ่าย: ${exp.toLocaleString()}</span><br><b>สุทธิ: ${(inc - exp).toLocaleString()}</b>`;
  }).getFilterByItemRange(item, mFrom, mTo, y);
}

    if (!data || data.length === 0) { area.style.display = 'none'; msg.style.display = 'block'; msg.innerHTML = `ไม่พบข้อมูลของ: <b>${m === 'all' ? 'ทุกเดือน' : mNames[m]} ปี ${y}</b>`; return; }
    msg.style.display = 'none'; area.style.display = 'block';
    document.getElementById('mHead').innerHTML = `จัดการข้อมูลของ: ${m === 'all' ? 'ทุกเดือน' : mNames[m]} ปี ${y}`;
    document.getElementById('mBody').innerHTML = data.map(r => `
      <tr>
        <td>${r.date}</td><td>${r.item}</td><td>${r.note}</td><td style="color:${r.type === 'รายรับ' ? 'green' : 'red'}">${r.amount.toLocaleString()}</td>
        <td>
          <div class="action-group">
            <button class="btn-sm" style="background:var(--edit)" onclick='openEdit(${JSON.stringify(r)})'>แก้</button>
            <button class="btn-sm" style="background:var(--danger)" onclick="delRow('${r.row}','${r.year}')">ลบ</button>
          </div>
        </td>
      </tr>`).join('');
  }).getRawData(m, y);
}

function openEdit(r) {
  document.getElementById('eRow').value = r.row;
  document.getElementById('eYear').value = r.year;
  if (r.rawDate) { document.getElementById('eDate').value = Array.isArray(r.rawDate) ? r.rawDate[0] : r.rawDate.split('T')[0]; }
  document.getElementById('eType').value = r.type;
  document.getElementById('eItem').value = r.item;
  document.getElementById('eAmt').value = r.amount;
  document.getElementById('eNote').value = r.note;
  document.getElementById('editModal').style.display = 'block';
}
function closeModal() { document.getElementById('editModal').style.display = 'none'; }

function delRow(r, y) { if(confirm('ยืนยันการลบรายการนี้?')) google.script.run.withSuccessHandler(res => { alert(res); loadM(); }).deleteRow(r,y); }
function toggleYearBox() { const b = document.getElementById('yearBox'); b.style.display = (b.style.display === 'none') ? 'block' : 'none'; }
function addNewYear() {

  if (!year || year.length < 4) return alert('กรุณาระบุปีให้ถูกต้อง');
  google.script.run.withSuccessHandler(res => { alert(res); updateYearLists(); toggleYearBox(); document.getElementById('newYearInp').value = ''; }).createNewYear(year);
}
